import pymysql

def get_conn():
    return pymysql.connect(
        host="localhost",
        user="root",
        password="root",
        database="shoes"
    )

def get_products():
    conn = get_conn()
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    cursor.execute("""
        select p.article, p.description,
        p.price, p.quantity, p.discount, p.photo,
        c.name as category, 
        t.name as type,
        m.name as manufacturer,
        s.name as supplier,
        u.name as unit
        from product p
        join category c on p.category_id = c.id
        join type t on p.type_id = t.id
        join manufacturer m on p.manufacturer_id = m.id
        join supplier s on p.supplier_id = s.id
        join unit u on p.unit_id = u.id
        """)
    products = cursor.fetchall()
    conn.close()
    return products

def get_orders():
    conn = get_conn()
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    cursor.execute("""
            select o.id, o.date_z, o.date_d, o.code, 
            os.name as status,
            pp.address as pickup_point
            from `order` o
            join order_status os on o.status_id = os.id
            join pickup_point pp on o.pickup_point_id = pp.id
            """)
    orders = cursor.fetchall()
    conn.close()
    return orders

class LoginWindow(QtWidgets.QMainWindow):
    def __init__(self):
        super().__init__()
        uic.loadUi("forms/login.ui", self)
        self.btnLogin.clicked.connect(self.login)
        self.btnGuest.clicked.connect(self.guest)

        pixmap = QPixmap("images/Icon.png")
        self.label_logo.setPixmap(pixmap)
        self.label_logo.setScaledContents(True)

    def login(self):
        login = self.inputLogin.text().strip()
        password = self.inputPassword.text().strip()

        if not login or not password:
            QtWidgets.QMessageBox.warning(self, "Ошибка", "Введите логин или пароль")
            return

        try:
            conn = get_conn()
            cursor = conn.cursor(pymysql.cursors.DictCursor)
            cursor.execute("""
                select u.id, r.name as role
                from user u 
                join role r on u.role_id = r.id
                where u.login = %s and u.password = %s
                """, (login, password))
            user = cursor.fetchone()
            conn.close()

            if not user:
                QtWidgets.QMessageBox.warning(self, "Ошибка", "Неверный логин или пароль")
                return

            self.open_by_role(user)

        except Exception as e:
            QtWidgets.QMessageBox.critical(self, "Ошибка", str(e))
            return

    def guest(self):
        from views.guest import GuestWindow
        self.window = GuestWindow()
        self.window.show()
        self.close()

    def open_by_role(self, user):
        role = user['role']
        if role == 'Администратор':
            from views.admin import AdminWindow
            self.window = AdminWindow(user['id'])
        elif role == 'Менеджер':
            from views.manager import ManagerWindow
            self.window = ManagerWindow(user['id'])
        elif role == 'Авторизированный клиент':
            from views.client import ClientWindow
            self.window = ClientWindow(user['id'])
        else:
            QtWidgets.QMessageBox.warning(self, "Ошибка", "Неверная роль")
            return
        self.window.show()
        self.close()

class OrdersWindow(QtWidgets.QMainWindow):
    def __init__(self):
        super().__init__()
        uic.loadUi("forms/client.ui", self)
        self.setWindowTitle("Заказы")
        self.labelUserName.setVisible(False)
        self.btnExit.clicked.connect(self.close)
        self.fill_list()

    def fill_list(self):
        orders = get_orders()
        self.listWidget.clear()
        for o in orders:
            card = self.create_card(o)
            item = QListWidgetItem()
            item.setSizeHint(card.sizeHint())
            self.listWidget.addItem(item)
            self.listWidget.setItemWidget(item, card)

    def create_card(self, o):
        card = QWidget()
        row = QHBoxLayout(card)

        info = QVBoxLayout()
        info.addWidget(QLabel(f"<b>Артикул заказа: {o['code']}</b>"))
        info.addWidget(QLabel(f"Статус: {o['status']}"))
        info.addWidget(QLabel(f"Адрес пункта выдачи: {o['pickup_point']}"))
        info.addWidget(QLabel(f"Дата заказа: {o['date_z']}"))
        row.addLayout(info)

        date_label = QLabel(f"Дата доставки:\n{o['date_d']}")
        date_label.setAlignment(Qt.AlignmentFlag.AlignCenter)
        date_label.setFixedWidth(120)
        row.addWidget(date_label)

    import pymysql
    from PyQt6 import QtWidgets, uic
    from PyQt6.QtWidgets import QListWidgetItem, QWidget, QLabel, QVBoxLayout, QHBoxLayout

    class AdminWindow(QtWidgets.QMainWindow):
        def __init__(self, user_id):
            super().__init__()
            uic.loadUi("forms/admin.ui", self)
            self.user_id = user_id

            self.inputSearch.textChanged.connect(self.apply_filters)
            self.comboSupplier.currentTextChanged.connect(self.apply_filters)
            self.btnSortAsc.clicked.connect(self.sort_asc)
            self.btnSortDesc.clicked.connect(self.sort_desc)
            self.btnExit.clicked.connect(self.logout)
            self.current_article = None
            self.btnDelete.clicked.connect(self.delete_product)
            self.btnOrders.clicked.connect(self.open_orders)
            self.listWidget.currentItemChanged.connect(self.on_item_selected)

            self.load_user_name()
            self.load_suppliers()
            self.fill_list()

        def load_user_name(self):
            conn = get_conn()
            cursor = conn.cursor(pymysql.cursors.DictCursor)
            cursor.execute(
                "select last_name, first_name, patronymic from user where id = %s",
                (self.user_id,)
            )
            user = cursor.fetchone()
            conn.close()
            self.labelUserName.setText(
                f"{user['last_name']} {user['first_name']} {user['patronymic']}"
            )

        def load_suppliers(self):
            conn = get_conn()
            cursor = conn.cursor(pymysql.cursors.DictCursor)
            cursor.execute("select name from supplier")
            suppliers = cursor.fetchall()
            conn.close()
            self.comboSupplier.addItem("Все поставщики")
            for s in suppliers:
                self.comboSupplier.addItem(s['name'])

        def fill_list(self, products=None):
            if products is None:
                products = get_products()
            self.products_on_screen = products
            self.listWidget.clear()
            for p in products:
                card = self.create_card(p)
                item = QListWidgetItem()
                item.setSizeHint(card.sizeHint())
                self.listWidget.addItem(item)
                self.listWidget.setItemWidget(item, card)

        def create_card(self, p):
            card = QWidget()
            row = QHBoxLayout(card)
            row.addWidget(self.make_photo(p))
            row.addLayout(self.make_info(p))
            row.addWidget(self.make_discount(p))
            self.apply_color(card, p)
            return card

        def make_photo(self, p):
            label = QLabel()
            path = f"images/{p['photo']}" if p['photo'] else "images/picture.png"
            label.setPixmap(QPixmap(path).scaled(80, 80, Qt.AspectRatioMode.KeepAspectRatio))
            label.setFixedWidth(90)
            return label

        def make_discount(self, p):
            label = QLabel(f"{p['discount']}%")
            label.setFixedWidth(60)
            label.setAlignment(Qt.AlignmentFlag.AlignCenter)
            return label

        def apply_color(self, card, p):
            if p['discount'] > 15:
                card.setStyleSheet("QWidget { background-color: #2E8B57; }")
            elif p['quantity'] == 0:
                card.setStyleSheet("QWidget { background-color: lightblue; }")

        def make_info(self, p):
            info = QVBoxLayout()
            info.addWidget(QLabel(f"<b>{p['category']} | {p['type']}</b>"))
            info.addWidget(QLabel(f"Описание: {p['description']}"))
            info.addWidget(QLabel(f"Производитель: {p['manufacturer']}"))
            info.addWidget(QLabel(f"Поставщик: {p['supplier']}"))
            info.addLayout(self.make_price(p))
            info.addWidget(QLabel(f"Ед.изм.: {p['unit']}"))
            info.addWidget(QLabel(f"Кол-во: {p['quantity']}"))
            return info

        def make_price(self, p):
            price_row = QHBoxLayout()
            price = p['price']
            discount = p['discount']
            if discount > 0:
                old = QLabel(f"{price} руб.")
                old.setStyleSheet("text-decoration: line-through; color: red;")
                price_row.addWidget(old)
                price_row.addWidget(QLabel(f"{round(price * (1 - discount / 100), 2)} руб."))
            else:
                price_row.addWidget(QLabel(f"{price} руб."))
            return price_row

        def apply_filters(self):
            products = get_products()

            supplier = self.comboSupplier.currentText()
            if supplier != "Все поставщики":
                products = [p for p in products if p['supplier'] == supplier]

            text = self.inputSearch.text().strip().lower()
            if text:
                products = [p for p in products if text in
                            (p['article'] + p['type'] + p['category'] + p['description'] +
                             p['manufacturer'] + p['supplier']).lower()
                            ]

            if hasattr(self, 'sort_order'):
                products.sort(
                    key=lambda p: p['quantity'],
                    reverse=(self.sort_order == 'desc')
                )

            self.fill_list(products)

        def sort_asc(self):
            self.sort_order = 'asc'
            self.apply_filters()

        def sort_desc(self):
            self.sort_order = 'desc'
            self.apply_filters()

        def on_item_selected(self, current, _):
            if current is None:
                self.current_article = None
                return
            self.current_article = self.products_on_screen[self.listWidget.row(current)]['article']

        def delete_product(self):
            if self.current_article is None:
                QtWidgets.QMessageBox.warning(self, "Ошибка", "Выберите товар для удаления.")
                return
            conn = get_conn()
            cursor = conn.cursor()
            cursor.execute("SELECT COUNT(*) FROM order_item WHERE article = %s", (self.current_article,))
            if cursor.fetchone()[0] > 0:
                conn.close()
                QtWidgets.QMessageBox.warning(self, "Ошибка", "Нельзя удалить товар, который присутствует в заказе.")
                return
            conn.close()
            reply = QtWidgets.QMessageBox.question(self, "Подтверждение", f"Удалить товар {self.current_article}?",
                                                   QtWidgets.QMessageBox.StandardButton.Yes | QtWidgets.QMessageBox.StandardButton.No)
            if reply != QtWidgets.QMessageBox.StandardButton.Yes:
                return
            conn = get_conn()
            cursor = conn.cursor()
            cursor.execute("DELETE FROM product WHERE article = %s", (self.current_article,))
            conn.commit()
            conn.close()
            self.current_article = None
            self.apply_filters()

        def open_orders(self):
            from views.orders import OrdersWindow
            self.orders_window = OrdersWindow()
            self.orders_window.show()

        def logout(self):
            from views.login import LoginWindow
            self.window = LoginWindow()
            self.window.show()
            self.close()
